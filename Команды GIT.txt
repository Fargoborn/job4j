 
mkdir <имя папки> //Команда создания папки

type NUL > <имя папки\name.ext> //Создание файла

mvn validate //проверка стиля кода

git status //проверка состояния репозитория

git add <имя папки\name.ext> //добавление папок/файлов в репозиторий (git add * //добавить все в текущей папке)

git commit -m "комментарий" //коммит изменений

git show HEAD //проверка результатов коммит

git fetch & git pull //синхронизация локального репозитория с репозиторием GitHab

git push origin master //проталкивание изменений в репозиторий GitHab

//удаление коммитов

1. Получаем хэш-код коммита, к которому хотим вернуться.
2. Заходим в папку репозитория и пишем в консоль:

git reset --hard хэш-код
git push --force
